ALTER Function [dbo].[UltimoDiaMes](@Date DateTime)

Returns DateTime

AS

Begin
	
	Set @Date = DateAdd(Month,1,@Date)

	Set @Date = Cast(Cast(Month(@Date)As varchar(2))+'-1-'+Cast(Year(@Date)As varchar(4)) As DateTime)

	Return DateAdd(Day,-1,@Date)
End
